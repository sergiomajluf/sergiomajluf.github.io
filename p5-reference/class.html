// Declare and construct two objects (h1, h2) from the class HLine 
var h1 = new HLine(20, 2.0);
var h2 = new HLine(50, 2.5);

function setup() {
    createCanvas(100, 100);
    frameRate(30);
}

function draw() {
    background(204);
    h1.update();
    h2.update();
}

function HLine(y, s) {
    var ypos = y;
    var speed = s;
    
    this.update = function() {
        ypos += speed;
        if (ypos > height) {
            ypos = 0;
        }
        line(0, ypos, width, ypos);
    }
}
