// Click on the image to give it focus,
// and then press any key.

// Note: The rectangle in this example may 
// flicker as the operating system may 
// register a long key press as a repetition
// of key presses.

function draw() {
    if (keyIsPressed()) {
        fill(0);
    } else {
        fill(255);
    }
    rect(25, 25, 50, 50);
}
