// Fill white on UP; black on down, grey everything else
var fillVal = 126;

function setup() {
    createCanvas(100, 100);        
}

function draw() {
  fill(fillVal);
  rect(25, 25, 50, 50);
}

function keyPressed() {
 // Get key CODES http://www.cambiaresearch.com/articles/15/javascript-char-codes-key-codes
    if (keyCode == 38) {
        fillVal = 255;
    } else if (keyCode == 40) {
        fillVal = 0;
    } else {
        fillVal = 126;
    }
}
